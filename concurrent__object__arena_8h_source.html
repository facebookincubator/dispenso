81  ConcurrentObjectArena(const ConcurrentObjectArena<T, Index, alignment>& other)
82  : kLog2BuffSize(other.kLog2BuffSize),
83  kBufferSize(other.kBufferSize),

157  Index grow_by(const Index delta) {
158  Index newPos;
159  Index oldPos = pos_.load(std::memory_order_relaxed);
160 
161  do {
162  Index curSize = allocatedSize_.load(std::memory_order_acquire);
163 
164  if (oldPos + delta >= curSize) {
165  const std::lock_guard<std::mutex> guard(resizeMutex_);
166  curSize = allocatedSize_.load(std::memory_order_relaxed);
167  while (oldPos + delta >= curSize) {
168  allocateBuffer();
169  allocatedSize_.store(curSize + kBufferSize, std::memory_order_release);
170  curSize = curSize + kBufferSize;
171  }
172  }
173 
174  newPos = oldPos + delta;
175  } while (!std::atomic_compare_exchange_weak_explicit(
176  &pos_, &oldPos, newPos, std::memory_order_release, std::memory_order_relaxed));
177 
178  constructObjects(oldPos, oldPos + delta);
179 
180  return oldPos;
181  }

302  void allocateBuffer() {
303  void* ptr = detail::alignedMalloc(kBufferSize * sizeof(T), alignment);
304 #if defined(__cpp_exceptions)
305  if (ptr == nullptr)
306  throw std::bad_alloc();
307 #endif // __cpp_exceptions
308 
309  if (buffersPos_ < buffersSize_) {
310  buffers_.load(std::memory_order_relaxed)[buffersPos_++] = static_cast<T*>(ptr);
311  } else {
312  const Index oldBuffersSize = buffersSize_;
313  T** oldBuffers = buffers_.load(std::memory_order_relaxed);
314 
315  buffersSize_ = oldBuffersSize == 0 ? 2 : oldBuffersSize * 2;
316  T** newBuffers = new T*[buffersSize_];
317 
318  if (oldBuffers != nullptr) {

320  deleteLater_.push_back(oldBuffers);

328  void constructObjects(const Index beginIndex, const Index endIndex) {
329  const Index startBuffer = beginIndex >> kLog2BuffSize;
330  const Index endBuffer = endIndex >> kLog2BuffSize;
331 
332  Index bufStart = beginIndex & kMask;
333  for (Index b = startBuffer; b <= endBuffer; ++b) {
334  T* buf = buffers_.load(std::memory_order_relaxed)[b];
335  const Index bufEnd = b == endBuffer ? (endIndex & kMask) : kBufferSize;
336  for (Index i = bufStart; i < bufEnd; ++i)
337  new (buf + i) T();
338 
339  bufStart = 0;
340  }
341  }
