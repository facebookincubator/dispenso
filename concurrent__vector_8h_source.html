412 void clear() {
413 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
414 size_t len = binfo.bucketIndex;
415 size_t cap = binfo.bucketCapacity;
416 size_t b = binfo.bucket;
417 do {
418 T* buf = buffers_[b].load(std::memory_order_relaxed);
419 T* t = buf + len;
420
421 while (t != buf) {
422 --t;
423 t->~T();
424 }
425 cap >>= int{b > 1};
426 len = cap;
427 } while (b--);
428 size_.store(0, std::memory_order_release);
429 }

435 void shrink_to_fit() {
436 constexpr size_t kMaxExtra = 2;
437 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
438
439 // We need to at least skip the first two buckets, since we have those as a single allocation.
440 size_t startBucket = std::max<size_t>(2, binfo.bucket + kMaxExtra);
441
442 for (size_t b = startBucket; b < kMaxBuffers; ++b) {
443 T* ptr = buffers_[b].load(std::memory_order_relaxed);
444 if (!ptr) {
445 break;
446 }
447 if (buffers_.shouldDealloc(b)) {
448 cv::dealloc<T>(ptr);
449 }
450 buffers_[b].store(nullptr, std::memory_order_release);
451 }
452 }

537 iterator erase(const_iterator pos) {
538 auto e = end();
539 if (e == pos) {
540 return e;
541 }
542 size_.fetch_sub(1, std::memory_order_relaxed);
543 --e;
544 if (e == pos) {
545 e->~T();
546 return e;
547 }
548 ++e;
549 auto it = begin();
550 it += (pos - it);
551 return std::move(pos + 1, const_iterator(e), it);
552 }

562 iterator erase(const_iterator first, const_iterator last) {
563 size_t len = std::distance(first, last);
564 auto it = begin();
565 size_t startIdx = first - it;
566 if (len == 0) {
567 return it + (startIdx + len);
568 }
569 it += startIdx;
570
571 auto e_it = std::move(last, cend(), it);
572
573 if (e_it < last) {
574 // remove any values that were not already moved into
575 do {
576 --last;
577 last->~T();
578 } while (e_it != last);
579 }
580 size_.fetch_sub(len, std::memory_order_relaxed);
581 return e_it;
582 }

610 auto binfo = bucketAndSubIndex(index);

989 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndexForIndex(size_t index) const {
990#if defined(__clang__)
991 if (index < firstBucketLen_) {
992 return {0, index, firstBucketLen_};
993 }
994
995 size_t l2idx = detail::log2(index);
996 size_t bucket = (l2idx + 1) - firstBucketShift_;
997 size_t bucketCapacity = size_t{1} << l2idx;
998 size_t bucketIndex = index - bucketCapacity;
999
1000 return {bucket, bucketIndex, bucketCapacity};
1001#else
1002 return bucketAndSubIndex(index);
1003#endif // __clang__
1004 }

1006 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndex(size_t index) const {
1007 size_t l2idx = detail::log2(index | 1);
1008 size_t bucket = (l2idx + 1) - firstBucketShift_;
1009 size_t bucketCapacity = size_t{1} << l2idx;
1010 size_t bucketIndex = index - bucketCapacity;
1011
1012 bucket = index < firstBucketLen_ ? 0 : bucket;
1013 bucketIndex = index < firstBucketLen_ ? index : bucketIndex;
1014 bucketCapacity = index < firstBucketLen_ ? firstBucketLen_ : bucketCapacity;
1015
1016 return {bucket, bucketIndex, bucketCapacity};
