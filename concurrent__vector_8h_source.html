397 size_t cap = 2 * firstBucketLen_;
398 for (size_t b = 2; b < kMaxBuffers; ++b) {

411 void clear() {
412 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
413 size_t len = binfo.bucketIndex;
414 size_t cap = binfo.bucketCapacity;
415 size_t b = binfo.bucket;
416 do {
417 T* buf = buffers_[b].load(std::memory_order_relaxed);
418 T* t = buf + len;
419
420 while (t != buf) {
421 --t;
422 t->~T();
423 }
424 cap >>= int{b > 1};
425 len = cap;
426 } while (b--);
427 size_.store(0, std::memory_order_release);
428 }

434 void shrink_to_fit() {
435 constexpr size_t kMaxExtra = 2;
436 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
437
438 // We need to at least skip the first two buckets, since we have those as a single allocation.
439 size_t startBucket = std::max<size_t>(2, binfo.bucket + kMaxExtra);
440
441 for (size_t b = startBucket; b < kMaxBuffers; ++b) {
442 T* ptr = buffers_[b].load(std::memory_order_relaxed);
443 if (!ptr) {
444 break;
445 }
446 if (buffers_.shouldDealloc(b)) {
447 cv::dealloc<T>(ptr);
448 }
449 buffers_[b].store(nullptr, std::memory_order_release);
450 }
451 }

536 iterator erase(const_iterator pos) {
537 auto e = end();
538 if (e == pos) {
539 return e;
540 }
541 size_.fetch_sub(1, std::memory_order_relaxed);
542 --e;
543 if (e == pos) {
544 e->~T();
545 return e;
546 }
547 ++e;
548 auto it = begin();
549 it += (pos - it);
550 return std::move(pos + 1, const_iterator(e), it);
551 }
552
561 iterator erase(const_iterator first, const_iterator last) {
562 size_t len = std::distance(first, last);
563 auto it = begin();
564 size_t startIdx = first - it;
565 if (len == 0) {
566 return it + (startIdx + len);
567 }
568 it += startIdx;
569
570 auto e_it = std::move(last, cend(), it);
571
572 if (e_it < last) {
573 // remove any values that were not already moved into
574 do {
575 --last;
576 last->~T();
577 } while (e_it != last);
578 }
579 size_.fetch_sub(len, std::memory_order_relaxed);
580 return e_it;
581 }

606 template <typename... Args>
607 iterator emplace_back(Args&&... args) {
608 auto index = size_.fetch_add(1, std::memory_order_relaxed);
609 auto binfo = bucketAndSubIndex(index);

613 iterator ret{this, index, binfo};

615 if (Traits::kIteratorPreferSpeed) {
616 new (&*ret) T(std::forward<Args>(args)...);

618 new (buffers_[binfo.bucket] + binfo.bucketIndex) T(std::forward<Args>(args)...);
619 }

988 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndexForIndex(size_t index) const {
989#if defined(__clang__)
990 if (index < firstBucketLen_) {
991 return {0, index, firstBucketLen_};
992 }
993
994 size_t l2idx = detail::log2(index);
995 size_t bucket = (l2idx + 1) - firstBucketShift_;
996 size_t bucketCapacity = size_t{1} << l2idx;
997 size_t bucketIndex = index - bucketCapacity;
998
999 return {bucket, bucketIndex, bucketCapacity};
1000#else
1001 return bucketAndSubIndex(index);
1002#endif // __clang__
1003 }

1005 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndex(size_t index) const {
1006 size_t l2idx = detail::log2(index | 1);
1007 size_t bucket = (l2idx + 1) - firstBucketShift_;
1008 size_t bucketCapacity = size_t{1} << l2idx;
1009 size_t bucketIndex = index - bucketCapacity;
1010
1011 bucket = index < firstBucketLen_ ? 0 : bucket;
1012 bucketIndex = index < firstBucketLen_ ? index : bucketIndex;
1013 bucketCapacity = index < firstBucketLen_ ? firstBucketLen_ : bucketCapacity;
1014
1015 return {bucket, bucketIndex, bucketCapacity};
1016 }
