394 size_type capacity() const {
395 size_t cap = 2 * firstBucketLen_;
396 for (size_t b = 2; b < kMaxBuffers; ++b) {
397 if (!buffers_[b].load(std::memory_order_relaxed)) {
398 break;
399 }

401 }

409 void clear() {
410 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
411 size_t len = binfo.bucketIndex;
412 size_t cap = binfo.bucketCapacity;
413 size_t b = binfo.bucket;
414 do {
415 T* buf = buffers_[b].load(std::memory_order_relaxed);
416 T* t = buf + len;
417
418 while (t != buf) {
419 --t;
420 t->~T();
421 }
422 cap >>= int{b > 1};
423 len = cap;
424 } while (b--);
425 size_.store(0, std::memory_order_release);
426 }

432 void shrink_to_fit() {
433 constexpr size_t kMaxExtra = 2;
434 auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
435
436 // We need to at least skip the first two buckets, since we have those as a single allocation.
437 size_t startBucket = std::max<size_t>(2, binfo.bucket + kMaxExtra);
438
439 for (size_t b = startBucket; b < kMaxBuffers; ++b) {
440 T* ptr = buffers_[b].load(std::memory_order_relaxed);
441 if (!ptr) {
442 break;
443 }
444 if (buffers_.shouldDealloc(b)) {
445 cv::dealloc<T>(ptr);
446 }
447 buffers_[b].store(nullptr, std::memory_order_release);
448 }
449 }

534 iterator erase(const_iterator pos) {
535 auto e = end();
536 if (e == pos) {
537 return e;
538 }
539 size_.fetch_sub(1, std::memory_order_relaxed);
540 --e;
541 if (e == pos) {
542 e->~T();
543 return e;
544 }
545 ++e;
546 auto it = begin();
547 it += (pos - it);
548 return std::move(pos + 1, const_iterator(e), it);
549 }
550
559 iterator erase(const_iterator first, const_iterator last) {
560 size_t len = std::distance(first, last);
561 auto it = begin();
562 size_t startIdx = first - it;
563 if (len == 0) {
564 return it + (startIdx + len);
565 }
566 it += startIdx;
567
568 auto e_it = std::move(last, cend(), it);
569
570 if (e_it < last) {
571 // remove any values that were not already moved into
572 do {
573 --last;
574 last->~T();
575 } while (e_it != last);
576 }
577 size_.fetch_sub(len, std::memory_order_relaxed);
578 return e_it;
579 }

604 template <typename... Args>
605 iterator emplace_back(Args&&... args) {
606 auto index = size_.fetch_add(1, std::memory_order_relaxed);
607 auto binfo = bucketAndSubIndex(index);
608
609 buffers_.allocAsNecessary(binfo);
610
611 iterator ret{this, index, binfo};
612
613 if (Traits::kIteratorPreferSpeed) {
614 new (&*ret) T(std::forward<Args>(args)...);
615 } else {
616 new (buffers_[binfo.bucket] + binfo.bucketIndex) T(std::forward<Args>(args)...);
617 }
618
619 return ret;

986 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndexForIndex(size_t index) const {
987#if defined(__clang__)
988 if (index < firstBucketLen_) {
989 return {0, index, firstBucketLen_};
990 }
991
992 size_t l2idx = detail::log2(index);
993 size_t bucket = (l2idx + 1) - firstBucketShift_;
994 size_t bucketCapacity = size_t{1} << l2idx;
995 size_t bucketIndex = index - bucketCapacity;
996
997 return {bucket, bucketIndex, bucketCapacity};
998#else
999 return bucketAndSubIndex(index);
1000#endif // __clang__
1001 }

1003 DISPENSO_INLINE cv::BucketInfo bucketAndSubIndex(size_t index) const {
1004 size_t l2idx = detail::log2(index | 1);
1005 size_t bucket = (l2idx + 1) - firstBucketShift_;
1006 size_t bucketCapacity = size_t{1} << l2idx;
1007 size_t bucketIndex = index - bucketCapacity;
1008
1009 bucket = index < firstBucketLen_ ? 0 : bucket;
1010 bucketIndex = index < firstBucketLen_ ? index : bucketIndex;
1011 bucketCapacity = index < firstBucketLen_ ? firstBucketLen_ : bucketCapacity;
1012
1013 return {bucket, bucketIndex, bucketCapacity};
1014 }
