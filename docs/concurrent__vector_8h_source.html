406  void clear() {
407  auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
408  size_t len = binfo.bucketIndex;
409  size_t cap = binfo.bucketCapacity;
410  size_t b = binfo.bucket;
411  do {
412  T* buf = buffers_[b].load(std::memory_order_relaxed);
413  T* t = buf + len;
414 
415  while (t != buf) {
416  --t;
417  t->~T();

419  cap >>= int{b > 1};
420  len = cap;
421  } while (b--);

429  void shrink_to_fit() {
430  constexpr size_t kMaxExtra = 2;
431  auto binfo = bucketAndSubIndex(size_.load(std::memory_order_relaxed));
432 
433  // We need to at least skip the first two buckets, since we have those as a single allocation.
434  size_t startBucket = std::max<size_t>(2, binfo.bucket + kMaxExtra);
435 
436  for (size_t b = startBucket; b < kMaxBuffers; ++b) {
437  T* ptr = buffers_[b].load(std::memory_order_relaxed);
438  if (!ptr) {
439  break;
440  }
441  if (buffers_.shouldDealloc(b)) {
442  cv::dealloc<T>(ptr);
443  }
444  buffers_[b].store(nullptr, std::memory_order_release);
445  }
446  }

984 #if defined(__clang__)

1000  DISPENSO_INLINE cv::BucketInfo bucketAndSubIndex(size_t index) const {
1001  size_t l2idx = detail::log2(index | 1);
1002  size_t bucket = (l2idx + 1) - firstBucketShift_;
1003  size_t bucketCapacity = size_t{1} << l2idx;
1004  size_t bucketIndex = index - bucketCapacity;
1005 
1006  bucket = index < firstBucketLen_ ? 0 : bucket;
1007  bucketIndex = index < firstBucketLen_ ? index : bucketIndex;
1008  bucketCapacity = index < firstBucketLen_ ? firstBucketLen_ : bucketCapacity;
1009 
1010  return {bucket, bucketIndex, bucketCapacity};
1011  }
