54 template <typename TaskSetT, typename Iter, typename F>
55 void for_each_n(TaskSetT& tasks, Iter start, size_t n, F&& f, ForEachOptions options = {}) {
56  // TODO(bbudge): With options.maxThreads, we might want to allow a small fanout factor in
57  // recursive case?
58  if (!n || (n == 1 && options.wait) || !options.maxThreads ||
59  detail::PerPoolPerThreadInfo::isParForRecursive(&tasks.pool())) {
60  for (size_t i = 0; i < n; ++i) {
61  f(*start);
62  ++start;
63  }
64  if (options.wait) {
65  tasks.wait();
66  }
67  return;
68  }
69 
70  ssize_t numThreads = std::min<ssize_t>(tasks.numPoolThreads(), options.maxThreads);
71  // Reduce threads used if they exceed work to be done.
72  numThreads = std::min<ssize_t>(numThreads, n);
73 
74  auto chunking = detail::staticChunkSize(n, numThreads);
75  size_t chunkSize = chunking.ceilChunkSize;
76 
77  bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
78 
79  // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
80  ssize_t firstLoopLen = chunking.transitionTaskIndex - perfectlyChunked;
81 
82  ssize_t t;
83  for (t = 0; t < firstLoopLen; ++t) {
84  Iter next = start;
85  std::advance(next, chunkSize);
86  tasks.schedule([start, next, f]() {
87  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
88  for (Iter it = start; it != next; ++it) {
89  f(*it);
90  }
91  });
92  start = next;
93  }
94 
95  // Reduce the remaining chunk sizes by 1.
96  chunkSize -= !perfectlyChunked;
97  // Finish submitting all but the last item.
98  for (; t < numThreads - 1; ++t) {
99  Iter next = start;
100  std::advance(next, chunkSize);
101  tasks.schedule([start, next, f]() {
102  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
103  for (Iter it = start; it != next; ++it) {
104  f(*it);
105  }
106  });
107  start = next;
108  }
109 
110  Iter end = start;
111  std::advance(end, chunkSize);
112 
113  if (options.wait) {
114  for (Iter it = start; it != end; ++it) {
115  f(*it);
116  }
117  tasks.wait();
118  } else {
119  tasks.schedule(
120  [start, end, f]() {
121  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
122  for (Iter it = start; it != end; ++it) {
123  f(*it);
124  }
125  },
126  ForceQueuingTag());
127  }
128 }
