112  template <typename OtherInt>
113  IntegerT calcChunkSize(OtherInt numLaunched, bool oneOnCaller) const {
114  ssize_t workingThreads = static_cast<ssize_t>(numLaunched) + ssize_t{oneOnCaller};
115  assert(workingThreads > 1);
116 
117  if (!chunk) {
118  // TODO(bbudge): play with different load balancing factors for auto.
119  // IntegerT dynFactor = std::log2(1.0f + (end_ - start_) / (cyclesPerIndex_ *
120  // cyclesPerIndex_));
121  constexpr ssize_t dynFactor = 16;
122  const ssize_t chunks = dynFactor * workingThreads;
123  return static_cast<IntegerT>((size() + chunks) / chunks);
124  } else if (chunk == kStatic) {
125  // This should never be called. The static distribution versions of the parallel_for
126  // functions should be invoked instead.
127  std::abort();
128  }
129  return chunk;
130  }
131 
132  IntegerT start;
133  IntegerT end;
134  IntegerT chunk;
135 };

168 template <typename TaskSetT, typename IntegerT, typename F>
169 void parallel_for_staticImpl(
170  TaskSetT& taskSet,
171  const ChunkedRange<IntegerT>& range,
172  F&& f,
173  ParForOptions options) {
174  ssize_t numThreads = std::min<ssize_t>(taskSet.numPoolThreads(), options.maxThreads);
175  // Reduce threads used if they exceed work to be done.
176  numThreads = std::min<ssize_t>(numThreads, range.size());
177 
178  auto chunking = detail::staticChunkSize(range.size(), numThreads);
179  IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
180 
181  bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
182 
183  // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
184  ssize_t firstLoopLen = static_cast<ssize_t>(chunking.transitionTaskIndex) - perfectlyChunked;
185 
186  IntegerT start = range.start;
187  ssize_t t;
188  for (t = 0; t < firstLoopLen; ++t) {
189  IntegerT next = static_cast<IntegerT>(start + chunkSize);
190  taskSet.schedule([start, next, f]() {
191  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
192  f(start, next);
193  });
194  start = next;
195  }
196 
197  // Reduce the remaining chunk sizes by 1.
198  chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
199  // Finish submitting all but the last item.
200  for (; t < numThreads - 1; ++t) {
201  IntegerT next = static_cast<IntegerT>(start + chunkSize);
202  taskSet.schedule([start, next, f]() {
203  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
204  f(start, next);
205  });
206  start = next;
207  }
208 
209  if (options.wait) {
210  f(start, range.end);
211  taskSet.wait();
212  } else {
213  taskSet.schedule(
214  [start, end = range.end, f]() {
215  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
216  f(start, end);
217  },
218  ForceQueuingTag());
219  }
220 }

300 void parallel_for(
301  TaskSetT& taskSet,
302  const ChunkedRange<IntegerT>& range,
303  F&& f,
304  ParForOptions options = {}) {
305  if (range.empty()) {
306  if (options.wait) {
307  taskSet.wait();
308  }
309  return;
310  }
311  // TODO(bbudge): With options.maxThreads, we might want to allow a small fanout factor in
312  // recursive case?
313  if (!options.maxThreads || range.size() == 1 ||
314  detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
315  f(range.start, range.end);
316  if (options.wait) {
317  taskSet.wait();
318  }
319  return;
320  }
321 
322  if (range.isStatic()) {
323  detail::parallel_for_staticImpl(taskSet, range, std::forward<F>(f), options);
324  return;
325  }
326 
327  const ssize_t N = taskSet.numPoolThreads();
328  const bool useCallingThread = options.wait;
329  const ssize_t numToLaunch = std::min<ssize_t>(options.maxThreads, N - useCallingThread);
330 
331  if (numToLaunch == 1 && !useCallingThread) {
332  taskSet.schedule([range, f = std::move(f)]() { f(range.start, range.end); });
333  if (options.wait) {
334  taskSet.wait();
335  }
336  return;
337  } else if (numToLaunch == 0) {
338  f(range.start, range.end);
339  if (options.wait) {
340  taskSet.wait();
341  }
342  return;
343  }
344 
345  const IntegerT chunk = range.calcChunkSize(numToLaunch, useCallingThread);
346 
347  if (options.wait) {
348  alignas(kCacheLineSize) std::atomic<IntegerT> index(range.start);
349  auto worker = [end = range.end, &index, f = std::move(f), chunk]() {
350  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
351  while (true) {
352  IntegerT cur = index.fetch_add(chunk, std::memory_order_relaxed);
353  if (cur >= end) {
354  break;
355  }
356  f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
357  }
358  };
359 
360  using Function = decltype(worker);
361 
362  for (ssize_t i = 0; i < numToLaunch; ++i) {
363  taskSet.schedule(Function(worker));
364  }
365  worker();
366  taskSet.wait();
367  } else {
368  struct Atomic {
369  Atomic(IntegerT i) : index(i) {}
370  char buffer[kCacheLineSize];
371  std::atomic<IntegerT> index;
372  char buffer2[kCacheLineSize];
373  };
374  // TODO(bbudge): dispenso::make_shared?
375  auto wrapper = std::make_shared<Atomic>(range.start);
376  auto worker = [end = range.end, wrapper = std::move(wrapper), f, chunk]() {
377  auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
378  while (true) {
379  IntegerT cur = wrapper->index.fetch_add(chunk, std::memory_order_relaxed);
380  if (cur >= end) {
381  break;
382  }
383  f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
384  }
385  };
386 
387  using Function = decltype(worker);
388 
389  for (ssize_t i = 0; i < numToLaunch; ++i) {
390  taskSet.schedule(Function(worker), ForceQueuingTag());
391  }
392  }
