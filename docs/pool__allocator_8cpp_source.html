24 char* PoolAllocator::alloc() {
25  while (true) {
26  uint32_t allocId = backingAllocLock_.fetch_add(1, std::memory_order_acquire);
27 
28  if (allocId == 0) {
29  if (chunks_.empty()) {
30  char* buffer = reinterpret_cast<char*>(allocFunc_(allocSize_));
31  backingAllocs_.push_back(buffer);
32  // Push n-1 values into the chunks_ buffer, and then return the nth.
33  for (size_t i = 0; i < chunksPerAlloc_ - 1; ++i) {
34  chunks_.push_back(buffer);
35  buffer += chunkSize_;
36  }
37  backingAllocLock_.store(0, std::memory_order_release);
38  return buffer;
39  }
40  char* back = chunks_.back();
41  chunks_.pop_back();
42  backingAllocLock_.store(0, std::memory_order_release);
43  return back;
44  } else {
45  std::this_thread::yield();
46  }
47  }
48 }

50 void PoolAllocator::dealloc(char* ptr) {
51  // For now do not release any memory back to the deallocFunc until destruction.
52  // TODO(bbudge): Consider cases where we haven't gotten below some threshold of ready chunks
53  // in a while. In that case, we could begin tracking allocations, and try to assemble entire
54  // starting allocations, possibly deferring a small amount to each alloc call. This would be
55  // slower, but would ensure we don't get into a situation where we need a bunch of memory up
56  // front, and then never again.
57 
58  while (true) {
59  uint32_t allocId = backingAllocLock_.fetch_add(1, std::memory_order_acquire);
60  if (allocId == 0) {
61  chunks_.push_back(ptr);
62  backingAllocLock_.store(0, std::memory_order_release);
63  break;
64  }
65  }
66 }
