16 void ThreadPool::threadLoop(std::atomic<bool>& running) {
17  using namespace std::chrono_literals;
18 
19  constexpr auto kSleepDuration = std::chrono::microseconds(DISPENSO_POLL_PERIOD_US);
20  constexpr int kBackoffYield = 50;
21  constexpr int kBackoffSleep = kBackoffYield + 5;
22  OnceFunction next;
23 
24  int failCount = 0;
25 
26  detail::PerPoolPerThreadInfo::registerPool(this);
27 
28  moodycamel::ConsumerToken token(work_);
29  while (running.load(std::memory_order_relaxed) ||
30  workRemaining_.load(std::memory_order_relaxed)) {
31  while (work_.try_dequeue(token, next)) {
32  executeNext(std::move(next));
33  failCount = 0;
34  }
35  ++failCount;
36  detail::cpuRelax();
37  if (failCount > kBackoffSleep) {
38  std::this_thread::sleep_for(kSleepDuration);
39  } else if (failCount > kBackoffYield) {
40  std::this_thread::yield();
41  }
42  }
43 }

45 void ThreadPool::resize(ssize_t sn) {
46  assert(sn > 0);
47  size_t n = static_cast<size_t>(sn);
48 
49  std::lock_guard<std::mutex> lk(threadsMutex_);
50  if (n < threads_.size()) {
51  for (size_t i = n; i < threads_.size(); ++i) {
52  threads_[i].running.store(false, std::memory_order_release);
53  }
54  for (size_t i = n; i < threads_.size(); ++i) {
55  threads_[i].thread.join();
56  }
57  while (threads_.size() > n) {
58  threads_.pop_back();
59  }
60  } else if (n > threads_.size()) {
61  for (size_t i = threads_.size(); i < n; ++i) {
62  threads_.emplace_back();
63  auto& back = threads_.back();
64  back.running = true;
65  back.thread = std::thread([this, &running = back.running]() { threadLoop(running); });
66  }
67  }
68  poolLoadFactor_.store(static_cast<ssize_t>(n * poolLoadMultiplier_), std::memory_order_relaxed);
69  numThreads_.store(sn, std::memory_order_relaxed);
70 }

79  std::unique_lock<std::mutex> lk(threadsMutex_, std::try_to_lock);
80  assert(lk.owns_lock());

84  }
85  while (tryExecuteNext()) {
86  }

88  t.thread.join();
89  }
