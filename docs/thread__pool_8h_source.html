179 inline void ThreadPool::schedule(F&& f) {
180  ssize_t curWork = workRemaining_.load(std::memory_order_relaxed);
181  ssize_t quickLoadFactor = numThreads_.load(std::memory_order_relaxed);
182  quickLoadFactor += quickLoadFactor / 2;
183  if ((detail::PerPoolPerThreadInfo::isPoolRecursive(this) && curWork > quickLoadFactor) ||
184  (curWork > poolLoadFactor_.load(std::memory_order_relaxed))) {
185  f();
186  } else {
187  schedule(std::move(f), ForceQueuingTag());
188  }
189 }
