86template <typename IntegerT = ssize_t>
87struct ChunkedRange {
88 // We need to utilize 64-bit integers to avoid overflow, e.g. passing -2**30, 2**30 as int32 will
89 // result in overflow unless we cast to 64-bit. Note that if we have a range of e.g. -2**63+1 to
90 // 2**63-1, we cannot hold the result in an int64_t. We could in a uint64_t, but it is quite
91 // tricky to make this work. However, I do not expect ranges larger than can be held in int64_t
92 // since people want their computations to finish before the heat death of the sun (slight
93 // exaggeration).
94 using size_type = std::conditional_t<std::is_signed<IntegerT>::value, int64_t, uint64_t>;
95
96 struct Static {};
97 struct Auto {};
98 static constexpr IntegerT kStatic = std::numeric_limits<IntegerT>::max();
99
107 ChunkedRange(IntegerT s, IntegerT e, IntegerT c) : start(s), end(e), chunk(c) {}
114 ChunkedRange(IntegerT s, IntegerT e, Static) : ChunkedRange(s, e, kStatic) {}
122 ChunkedRange(IntegerT s, IntegerT e, Auto) : ChunkedRange(s, e, 0) {}
123
124 bool isStatic() const {
125 return chunk == kStatic;
126 }
127
128 bool isAuto() const {
129 return chunk == 0;
130 }
131
132 bool empty() const {
133 return end <= start;
134 }
135
136 size_type size() const {
137 return static_cast<size_type>(end) - start;
138 }

140 template <typename OtherInt>
141 std::tuple<size_type, size_type>
142 calcChunkSize(OtherInt numLaunched, bool oneOnCaller, size_type minChunkSize) const {
143 size_type workingThreads = static_cast<size_type>(numLaunched) + size_type{oneOnCaller};
144 assert(workingThreads > 1);
145
146 if (!chunk) {
147 size_type dynFactor = std::min<size_type>(16, size() / workingThreads);
148 size_type chunkSize;
149 do {
150 size_type roughChunks = dynFactor * workingThreads;
151 chunkSize = (size() + roughChunks - 1) / roughChunks;
152 --dynFactor;
153 } while (chunkSize < minChunkSize);
154 return {chunkSize, (size() + chunkSize - 1) / chunkSize};
155 } else if (chunk == kStatic) {
156 // This should never be called. The static distribution versions of the parallel_for
157 // functions should be invoked instead.
158 std::abort();
159 }
160 return {chunk, (size() + chunk - 1) / chunk};
161 }
162
163 IntegerT start;
164 IntegerT end;
165 IntegerT chunk;
166};

241template <
242 typename TaskSetT,
243 typename IntegerT,
244 typename F,
245 typename StateContainer,
246 typename StateGen>
247void parallel_for_staticImpl(
248 TaskSetT& taskSet,
249 StateContainer& states,
250 const StateGen& defaultState,
251 const ChunkedRange<IntegerT>& range,
252 F&& f,
253 ssize_t maxThreads,
254 bool wait,
255 bool reuseExistingState) {
256 using size_type = typename ChunkedRange<IntegerT>::size_type;
257
258 size_type numThreads = std::min<size_type>(taskSet.numPoolThreads(), maxThreads);
259 // Reduce threads used if they exceed work to be done.
260 numThreads = std::min(numThreads, range.size()) + wait;
261
262 if (!reuseExistingState) {
263 states.clear();
264 }
265
266 size_t numToEmplace =
267 states.size() < static_cast<size_t>(numThreads) ? numThreads - states.size() : 0;
268
269 for (; numToEmplace--;) {
270 states.emplace_back(defaultState());
271 }
272
273 auto chunking = detail::staticChunkSize(range.size(), numThreads);
274 IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
275
276 bool perfectlyChunked = static_cast<size_type>(chunking.transitionTaskIndex) == numThreads;
277
278 // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
279 size_type firstLoopLen = chunking.transitionTaskIndex - perfectlyChunked;
280
281 auto stateIt = states.begin();
282 IntegerT start = range.start;
283 size_type t;
284 for (t = 0; t < firstLoopLen; ++t) {
285 IntegerT next = static_cast<IntegerT>(start + chunkSize);
286 taskSet.schedule([it = stateIt++, start, next, f]() {
287 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
288 f(*it, start, next);
289 });
290 start = next;
291 }
292
293 // Reduce the remaining chunk sizes by 1.
294 chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
295 // Finish submitting all but the last item.
296 for (; t < numThreads - 1; ++t) {
297 IntegerT next = static_cast<IntegerT>(start + chunkSize);
298 taskSet.schedule([it = stateIt++, start, next, f]() {
299 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
300 f(*it, start, next);
301 });
302 start = next;
303 }
304
305 if (wait) {
306 f(*stateIt, start, range.end);
307 taskSet.wait();
308 } else {
309 taskSet.schedule(
310 [stateIt, start, end = range.end, f]() {
311 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
312 f(*stateIt, start, end);
313 },
314 ForceQueuingTag());
315 }
316}

336template <
337 typename TaskSetT,
338 typename IntegerT,
339 typename F,
340 typename StateContainer,
341 typename StateGen>
342void parallel_for(
343 TaskSetT& taskSet,
344 StateContainer& states,
345 const StateGen& defaultState,
346 const ChunkedRange<IntegerT>& range,
347 F&& f,
348 ParForOptions options = {}) {
349 if (range.empty()) {
350 if (options.wait) {
351 taskSet.wait();
352 }
353 return;
354 }
355
356 using size_type = typename ChunkedRange<IntegerT>::size_type;
357
358 // Ensure minItemsPerChunk is sane
359 uint32_t minItemsPerChunk = std::max<uint32_t>(1, options.minItemsPerChunk);
360 size_type maxThreads = options.maxThreads;
361 bool isStatic = range.isStatic();
362
363 const size_type N = taskSet.numPoolThreads();
364 if (N == 0 || !options.maxThreads || range.size() <= minItemsPerChunk ||
365 detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
366 if (!options.reuseExistingState) {
367 states.clear();
368 }
369 if (states.empty()) {
370 states.emplace_back(defaultState());
371 }
372 f(*states.begin(), range.start, range.end);
373 if (options.wait) {
374 taskSet.wait();
375 }
376 return;
377 }
378
379 // Adjust down workers if we would have too-small chunks
380 if (minItemsPerChunk > 1) {
381 size_type maxWorkers = range.size() / minItemsPerChunk;
382 if (maxWorkers < maxThreads) {
383 maxThreads = static_cast<uint32_t>(maxWorkers);
384 }
385 if (range.size() / (maxThreads + options.wait) < minItemsPerChunk && range.isAuto()) {
386 isStatic = true;
387 }
388 } else if (range.size() <= N + options.wait) {
389 if (range.isAuto()) {
390 isStatic = true;
391 } else if (!range.isStatic()) {
392 maxThreads = range.size() - options.wait;
393 }
394 }
395
396 if (isStatic) {
397 detail::parallel_for_staticImpl(
398 taskSet,
399 states,
400 defaultState,
401 range,
402 std::forward<F>(f),
403 maxThreads,
404 options.wait,
405 options.reuseExistingState);
406 return;
407 }
408
409 const size_type numToLaunch = std::min<size_type>(maxThreads, N);
410
411 if (!options.reuseExistingState) {
412 states.clear();
413 }
414
415 size_t numToEmplace = static_cast<size_type>(states.size()) < (numToLaunch + options.wait)
416 ? (numToLaunch + options.wait) - states.size()
417 : 0;
418 for (; numToEmplace--;) {
419 states.emplace_back(defaultState());
420 }
421
422 if (numToLaunch == 1 && !options.wait) {
423 taskSet.schedule(
424 [&s = states.front(), range, f = std::move(f)]() { f(s, range.start, range.end); });
425
426 return;
427 }
428
429 auto chunkInfo = range.calcChunkSize(numToLaunch, options.wait, minItemsPerChunk);
430 auto chunkSize = std::get<0>(chunkInfo);
431 auto numChunks = std::get<1>(chunkInfo);
432
433 if (options.wait) {
434 alignas(kCacheLineSize) std::atomic<decltype(numChunks)> index(0);
435 auto worker = [start = range.start, end = range.end, &index, f, chunkSize, numChunks](auto& s) {
436 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
437
438 while (true) {
439 auto cur = index.fetch_add(1, std::memory_order_relaxed);
440 if (cur >= numChunks) {
441 break;
442 }
443 auto sidx = static_cast<IntegerT>(start + cur * chunkSize);
444 if (cur + 1 == numChunks) {
445 f(s, sidx, end);
446 } else {
447 auto eidx = static_cast<IntegerT>(sidx + chunkSize);
448 f(s, sidx, eidx);
449 }
450 }
451 };
452
453 auto it = states.begin();
454 for (size_type i = 0; i < numToLaunch; ++i) {
455 taskSet.schedule([&s = *it++, worker]() { worker(s); });
456 }
457 worker(*it);
458 taskSet.wait();
459 } else {
460 struct Atomic {
461 Atomic() : index(0) {}
462 alignas(kCacheLineSize) std::atomic<decltype(numChunks)> index;
463 char buffer[kCacheLineSize - sizeof(index)];
464 };
465 auto wrapper = std::make_shared<Atomic>();
466 auto worker = [start = range.start,
467 end = range.end,
468 wrapper = std::move(wrapper),
469 f,
470 chunkSize,
471 numChunks](auto& s) {
472 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
473 while (true) {
474 auto cur = wrapper->index.fetch_add(1, std::memory_order_relaxed);
475 if (cur >= numChunks) {
476 break;
477 }
478 auto sidx = static_cast<IntegerT>(start + cur * chunkSize);
479 if (cur + 1 == numChunks) {
480 f(s, sidx, end);
481 } else {
482 auto eidx = static_cast<IntegerT>(sidx + chunkSize);
483 f(s, sidx, eidx);
484 }
485 }
486 };
487
488 auto it = states.begin();
489 for (size_type i = 0; i < numToLaunch; ++i) {
490 taskSet.schedule([&s = *it++, worker]() { worker(s); }, ForceQueuingTag());
491 }
492 }
493}
