115 template <typename OtherInt>
116 IntegerT calcChunkSize(OtherInt numLaunched, bool oneOnCaller) const {
117 ssize_t workingThreads = static_cast<ssize_t>(numLaunched) + ssize_t{oneOnCaller};
118 assert(workingThreads > 1);
119
120 if (!chunk) {
121 // TODO(bbudge): play with different load balancing factors for auto.
122 // IntegerT dynFactor = std::log2(1.0f + (end_ - start_) / (cyclesPerIndex_ *
123 // cyclesPerIndex_));
124 constexpr ssize_t dynFactor = 16;
125 const ssize_t chunks = dynFactor * workingThreads;
126 return static_cast<IntegerT>((size() + chunks) / chunks);
127 } else if (chunk == kStatic) {
128 // This should never be called. The static distribution versions of the parallel_for
129 // functions should be invoked instead.
130 std::abort();
131 }
132 return chunk;
133 }
134
135 IntegerT start;
136 IntegerT end;
137 IntegerT chunk;
138};

171template <typename TaskSetT, typename IntegerT, typename F>
172void parallel_for_staticImpl(
173 TaskSetT& taskSet,
174 const ChunkedRange<IntegerT>& range,
175 F&& f,
176 ParForOptions options) {
177 ssize_t numThreads = std::min<ssize_t>(taskSet.numPoolThreads(), options.maxThreads);
178 // Reduce threads used if they exceed work to be done.
179 numThreads = std::min<ssize_t>(numThreads, range.size()) + options.wait;
180
181 auto chunking = detail::staticChunkSize(range.size(), numThreads);
182 IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
183
184 bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
185
186 // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
187 ssize_t firstLoopLen = static_cast<ssize_t>(chunking.transitionTaskIndex) - perfectlyChunked;
188
189 IntegerT start = range.start;
190 ssize_t t;
191 for (t = 0; t < firstLoopLen; ++t) {
192 IntegerT next = static_cast<IntegerT>(start + chunkSize);
193 taskSet.schedule([start, next, f]() {
194 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
195 f(start, next);
196 });
197 start = next;
198 }
199
200 // Reduce the remaining chunk sizes by 1.
201 chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
202 // Finish submitting all but the last item.
203 for (; t < numThreads - 1; ++t) {
204 IntegerT next = static_cast<IntegerT>(start + chunkSize);
205 taskSet.schedule([start, next, f]() {
206 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
207 f(start, next);
208 });
209 start = next;
210 }
211
212 if (options.wait) {
213 f(start, range.end);
214 taskSet.wait();
215 } else {
216 taskSet.schedule(
217 [start, end = range.end, f]() {
218 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
219 f(start, end);
220 },
221 ForceQueuingTag());
222 }
223}

225template <
226 typename TaskSetT,
227 typename IntegerT,
228 typename F,
229 typename StateContainer,
230 typename StateGen>
231void parallel_for_staticImpl(
232 TaskSetT& taskSet,
233 StateContainer& states,
234 const StateGen& defaultState,
235 const ChunkedRange<IntegerT>& range,
236 F&& f,
237 ParForOptions options) {
238 ssize_t numThreads = std::min<ssize_t>(taskSet.numPoolThreads(), options.maxThreads);
239 // Reduce threads used if they exceed work to be done.
240 numThreads = std::min<ssize_t>(numThreads, range.size()) + options.wait;
241
242 for (ssize_t i = 0; i < numThreads; ++i) {
243 states.emplace_back(defaultState());
244 }
245
246 auto chunking = detail::staticChunkSize(range.size(), numThreads);
247 IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
248
249 bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
250
251 // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
252 ssize_t firstLoopLen = chunking.transitionTaskIndex - perfectlyChunked;
253
254 auto stateIt = states.begin();
255 IntegerT start = range.start;
256 ssize_t t;
257 for (t = 0; t < firstLoopLen; ++t) {
258 IntegerT next = static_cast<IntegerT>(start + chunkSize);
259 taskSet.schedule([it = stateIt++, start, next, f]() {
260 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
261 f(*it, start, next);
262 });
263 start = next;
264 }
265
266 // Reduce the remaining chunk sizes by 1.
267 chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
268 // Finish submitting all but the last item.
269 for (; t < numThreads - 1; ++t) {
270 IntegerT next = static_cast<IntegerT>(start + chunkSize);
271 taskSet.schedule([it = stateIt++, start, next, f]() {
272 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
273 f(*it, start, next);
274 });
275 start = next;
276 }
277
278 if (options.wait) {
279 f(*stateIt, start, range.end);
280 taskSet.wait();
281 } else {
282 taskSet.schedule(
283 [stateIt, start, end = range.end, f]() {
284 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
285 f(*stateIt, start, end);
286 },
287 ForceQueuingTag());
288 }

302template <typename TaskSetT, typename IntegerT, typename F>
303void parallel_for(
304 TaskSetT& taskSet,
305 const ChunkedRange<IntegerT>& range,
306 F&& f,
307 ParForOptions options = {}) {
308 if (range.empty()) {
309 if (options.wait) {
310 taskSet.wait();
311 }
312 return;
313 }
314 // TODO(bbudge): With options.maxThreads, we might want to allow a small fanout factor in
315 // recursive case?
316 const ssize_t N = taskSet.numPoolThreads();
317 if (N == 0 || !options.maxThreads || range.size() == 1 ||
318 detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
319 f(range.start, range.end);
320 if (options.wait) {
321 // Note that we may have to wait here despite currently running single threaded because we may
322 // need to block prior work on the taskSet
323 taskSet.wait();
324 }
325 return;
326 }
327
328 if (range.isStatic()) {
329 detail::parallel_for_staticImpl(taskSet, range, std::forward<F>(f), options);
330 return;
331 }
332
333 const ssize_t numToLaunch = std::min<ssize_t>(options.maxThreads, N);
334
335 if (numToLaunch == 1 && !options.wait) {
336 taskSet.schedule([range, f = std::move(f)]() { f(range.start, range.end); });
337 return;
338 }
339
340 const IntegerT chunk = range.calcChunkSize(numToLaunch, options.wait);
341
342 if (options.wait) {
343 alignas(kCacheLineSize) std::atomic<IntegerT> index(range.start);
344 auto worker = [end = range.end, &index, f = std::move(f), chunk]() {
345 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
346 while (true) {
347 IntegerT cur = index.fetch_add(chunk, std::memory_order_relaxed);
348 if (cur >= end) {
349 break;
350 }
351 f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
352 }
353 };
354
355 using Function = decltype(worker);
356
357 for (ssize_t i = 0; i < numToLaunch; ++i) {
358 taskSet.schedule(Function(worker));
359 }
360 worker();
361 taskSet.wait();
362 } else {
363 struct Atomic {
364 Atomic(IntegerT i) : index(i) {}
365 char buffer[kCacheLineSize];
366 std::atomic<IntegerT> index;
367 char buffer2[kCacheLineSize];
368 };
369 // TODO(bbudge): dispenso::make_shared?
370 auto wrapper = std::make_shared<Atomic>(range.start);
371 auto worker = [end = range.end, wrapper = std::move(wrapper), f, chunk]() {
372 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
373 while (true) {
374 IntegerT cur = wrapper->index.fetch_add(chunk, std::memory_order_relaxed);
375 if (cur >= end) {
376 break;
377 }
378 f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
379 }
380 };
381
382 using Function = decltype(worker);
383
384 for (ssize_t i = 0; i < numToLaunch; ++i) {
385 taskSet.schedule(Function(worker), ForceQueuingTag());
386 }
387 }
388}

422template <
423 typename TaskSetT,
424 typename IntegerT,
425 typename F,
426 typename StateContainer,
427 typename StateGen>
428void parallel_for(
429 TaskSetT& taskSet,
430 StateContainer& states,
431 const StateGen& defaultState,
432 const ChunkedRange<IntegerT>& range,
433 F&& f,
434 ParForOptions options = {}) {
435 if (range.empty()) {
436 if (options.wait) {
437 taskSet.wait();
438 }
439 return;
440 }
441 const ssize_t N = taskSet.numPoolThreads();
442 if (N == 0 || !options.maxThreads || range.size() == 1 ||
443 detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
444 states.emplace_back(defaultState());
445 f(*states.begin(), range.start, range.end);
446 if (options.wait) {
447 taskSet.wait();
448 }
449 return;
450 }
451
452 if (range.isStatic()) {
453 detail::parallel_for_staticImpl(
454 taskSet, states, defaultState, range, std::forward<F>(f), options);
455 return;
456 }
457
458 const ssize_t numToLaunch = std::min<ssize_t>(options.maxThreads, N);
459
460 for (ssize_t i = 0; i < numToLaunch + options.wait; ++i) {
461 states.emplace_back(defaultState());
462 }
463
464 if (numToLaunch == 1 && !options.wait) {
465 taskSet.schedule(
466 [&s = states.front(), range, f = std::move(f)]() { f(s, range.start, range.end); });
467
468 return;
469 }
470
471 const IntegerT chunk = range.calcChunkSize(numToLaunch, options.wait);
472
473 if (options.wait) {
474 alignas(kCacheLineSize) std::atomic<IntegerT> index(range.start);
475 auto worker = [end = range.end, &index, f, chunk](auto& s) {
476 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
477
478 while (true) {
479 IntegerT cur = index.fetch_add(chunk, std::memory_order_relaxed);
480 if (cur >= end) {
481 break;
482 }
483 f(s, cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
484 }
485 };
486
487 auto it = states.begin();
488 for (ssize_t i = 0; i < numToLaunch; ++i) {
489 taskSet.schedule([&s = *it++, worker]() { worker(s); });
490 }
491 worker(*it);
492 taskSet.wait();
493 } else {
494 struct Atomic {
495 Atomic(IntegerT i) : index(i) {}
496 char buffer[kCacheLineSize];
497 std::atomic<IntegerT> index;
498 char buffer2[kCacheLineSize];
499 };
500 auto wrapper = std::make_shared<Atomic>(range.start);
501 auto worker = [end = range.end, wrapper = std::move(wrapper), f, chunk](auto& s) {
502 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
503 while (true) {
504 IntegerT cur = wrapper->index.fetch_add(chunk, std::memory_order_relaxed);
505 if (cur >= end) {
506 break;
507 }
508 f(s, cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
509 }
510 };
511
512 auto it = states.begin();
513 for (ssize_t i = 0; i < numToLaunch; ++i) {
514 taskSet.schedule([&s = *it++, worker]() { worker(s); }, ForceQueuingTag());
515 }
516 }
