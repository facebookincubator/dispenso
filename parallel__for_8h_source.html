113 template <typename OtherInt>
114 IntegerT calcChunkSize(OtherInt numLaunched, bool oneOnCaller) const {
115 ssize_t workingThreads = static_cast<ssize_t>(numLaunched) + ssize_t{oneOnCaller};
116 assert(workingThreads > 1);
117
118 if (!chunk) {
119 // TODO(bbudge): play with different load balancing factors for auto.
120 // IntegerT dynFactor = std::log2(1.0f + (end_ - start_) / (cyclesPerIndex_ *
121 // cyclesPerIndex_));
122 constexpr ssize_t dynFactor = 16;
123 const ssize_t chunks = dynFactor * workingThreads;
124 return static_cast<IntegerT>((size() + chunks) / chunks);
125 } else if (chunk == kStatic) {
126 // This should never be called. The static distribution versions of the parallel_for
127 // functions should be invoked instead.
128 std::abort();
129 }
130 return chunk;
131 }
132
133 IntegerT start;
134 IntegerT end;
135 IntegerT chunk;
136};

169template <typename TaskSetT, typename IntegerT, typename F>
170void parallel_for_staticImpl(
171 TaskSetT& taskSet,
172 const ChunkedRange<IntegerT>& range,
173 F&& f,
174 ParForOptions options) {
175 ssize_t numThreads = std::min<ssize_t>(taskSet.numPoolThreads(), options.maxThreads);
176 // Reduce threads used if they exceed work to be done.
177 numThreads = std::min<ssize_t>(numThreads, range.size());
178
179 auto chunking = detail::staticChunkSize(range.size(), numThreads);
180 IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
181
182 bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
183
184 // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
185 ssize_t firstLoopLen = static_cast<ssize_t>(chunking.transitionTaskIndex) - perfectlyChunked;
186
187 IntegerT start = range.start;
188 ssize_t t;
189 for (t = 0; t < firstLoopLen; ++t) {
190 IntegerT next = static_cast<IntegerT>(start + chunkSize);
191 taskSet.schedule([start, next, f]() {
192 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
193 f(start, next);
194 });
195 start = next;
196 }
197
198 // Reduce the remaining chunk sizes by 1.
199 chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
200 // Finish submitting all but the last item.
201 for (; t < numThreads - 1; ++t) {
202 IntegerT next = static_cast<IntegerT>(start + chunkSize);
203 taskSet.schedule([start, next, f]() {
204 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
205 f(start, next);
206 });
207 start = next;
208 }
209
210 if (options.wait) {
211 f(start, range.end);
212 taskSet.wait();
213 } else {
214 taskSet.schedule(
215 [start, end = range.end, f]() {
216 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
217 f(start, end);
218 },
219 ForceQueuingTag());
220 }
221}

223template <
224 typename TaskSetT,
225 typename IntegerT,
226 typename F,
227 typename StateContainer,
228 typename StateGen>
229void parallel_for_staticImpl(
230 TaskSetT& taskSet,
231 StateContainer& states,
232 const StateGen& defaultState,
233 const ChunkedRange<IntegerT>& range,
234 F&& f,
235 ParForOptions options) {
236 ssize_t numThreads = std::min<ssize_t>(taskSet.numPoolThreads(), options.maxThreads);
237 // Reduce threads used if they exceed work to be done.
238 numThreads = std::min<ssize_t>(numThreads, range.size());
239
240 for (ssize_t i = 0; i < numThreads; ++i) {
241 states.emplace_back(defaultState());
242 }
243
244 auto chunking = detail::staticChunkSize(range.size(), numThreads);
245 IntegerT chunkSize = static_cast<IntegerT>(chunking.ceilChunkSize);
246
247 bool perfectlyChunked = chunking.transitionTaskIndex == numThreads;
248
249 // (!perfectlyChunked) ? chunking.transitionTaskIndex : numThreads - 1;
250 ssize_t firstLoopLen = chunking.transitionTaskIndex - perfectlyChunked;
251
252 auto stateIt = states.begin();
253 IntegerT start = range.start;
254 ssize_t t;
255 for (t = 0; t < firstLoopLen; ++t) {
256 IntegerT next = static_cast<IntegerT>(start + chunkSize);
257 taskSet.schedule([it = stateIt++, start, next, f]() {
258 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
259 f(*it, start, next);
260 });
261 start = next;
262 }
263
264 // Reduce the remaining chunk sizes by 1.
265 chunkSize = static_cast<IntegerT>(chunkSize - !perfectlyChunked);
266 // Finish submitting all but the last item.
267 for (; t < numThreads - 1; ++t) {
268 IntegerT next = static_cast<IntegerT>(start + chunkSize);
269 taskSet.schedule([it = stateIt++, start, next, f]() {
270 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
271 f(*it, start, next);
272 });
273 start = next;
274 }
275
276 if (options.wait) {
277 f(*stateIt, start, range.end);
278 taskSet.wait();
279 } else {
280 taskSet.schedule(
281 [stateIt, start, end = range.end, f]() {
282 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
283 f(*stateIt, start, end);
284 },
285 ForceQueuingTag());
286 }
287}

300template <typename TaskSetT, typename IntegerT, typename F>
301void parallel_for(
302 TaskSetT& taskSet,
303 const ChunkedRange<IntegerT>& range,
304 F&& f,
305 ParForOptions options = {}) {
306 if (range.empty()) {
307 if (options.wait) {
308 taskSet.wait();
309 }
310 return;
311 }
312 // TODO(bbudge): With options.maxThreads, we might want to allow a small fanout factor in
313 // recursive case?
314 const ssize_t N = taskSet.numPoolThreads();
315 if (N == 0 || !options.maxThreads || range.size() == 1 ||
316 detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
317 f(range.start, range.end);
318 if (options.wait) {
319 taskSet.wait();
320 }
321 return;
322 }
323
324 if (range.isStatic()) {
325 detail::parallel_for_staticImpl(taskSet, range, std::forward<F>(f), options);
326 return;
327 }
328
329 const bool useCallingThread = options.wait;
330 const ssize_t numToLaunch = std::min<ssize_t>(options.maxThreads, N - useCallingThread);
331
332 if (numToLaunch == 1 && !useCallingThread) {
333 taskSet.schedule([range, f = std::move(f)]() { f(range.start, range.end); });
334 if (options.wait) {
335 taskSet.wait();
336 }
337 return;
338 } else if (numToLaunch == 0) {
339 f(range.start, range.end);
340 if (options.wait) {
341 taskSet.wait();
342 }
343 return;
344 }
345
346 const IntegerT chunk = range.calcChunkSize(numToLaunch, useCallingThread);
347
348 if (options.wait) {
349 alignas(kCacheLineSize) std::atomic<IntegerT> index(range.start);
350 auto worker = [end = range.end, &index, f = std::move(f), chunk]() {
351 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
352 while (true) {
353 IntegerT cur = index.fetch_add(chunk, std::memory_order_relaxed);
354 if (cur >= end) {
355 break;
356 }
357 f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
358 }
359 };
360
361 using Function = decltype(worker);
362
363 for (ssize_t i = 0; i < numToLaunch; ++i) {
364 taskSet.schedule(Function(worker));
365 }
366 worker();
367 taskSet.wait();
368 } else {
369 struct Atomic {
370 Atomic(IntegerT i) : index(i) {}
371 char buffer[kCacheLineSize];
372 std::atomic<IntegerT> index;
373 char buffer2[kCacheLineSize];
374 };
375 // TODO(bbudge): dispenso::make_shared?
376 auto wrapper = std::make_shared<Atomic>(range.start);
377 auto worker = [end = range.end, wrapper = std::move(wrapper), f, chunk]() {
378 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
379 while (true) {
380 IntegerT cur = wrapper->index.fetch_add(chunk, std::memory_order_relaxed);
381 if (cur >= end) {
382 break;
383 }
384 f(cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
385 }
386 };
387
388 using Function = decltype(worker);
389
390 for (ssize_t i = 0; i < numToLaunch; ++i) {
391 taskSet.schedule(Function(worker), ForceQueuingTag());
392 }
393 }

428template <
429 typename TaskSetT,
430 typename IntegerT,
431 typename F,
432 typename StateContainer,
433 typename StateGen>
434void parallel_for(
435 TaskSetT& taskSet,
436 StateContainer& states,
437 const StateGen& defaultState,
438 const ChunkedRange<IntegerT>& range,
439 F&& f,
440 ParForOptions options = {}) {
441 if (range.empty()) {
442 if (options.wait) {
443 taskSet.wait();
444 }
445 return;
446 }
447 const ssize_t N = taskSet.numPoolThreads();
448 if (N == 0 || !options.maxThreads || range.size() == 1 ||
449 detail::PerPoolPerThreadInfo::isParForRecursive(&taskSet.pool())) {
450 states.emplace_back(defaultState());
451 f(*states.begin(), range.start, range.end);
452 if (options.wait) {
453 taskSet.wait();
454 }
455 return;
456 }
457
458 if (range.isStatic()) {
459 detail::parallel_for_staticImpl(
460 taskSet, states, defaultState, range, std::forward<F>(f), options);
461 return;
462 }
463
464 const bool useCallingThread = options.wait;
465 const ssize_t numToLaunch = std::min<ssize_t>(options.maxThreads, N - useCallingThread);
466
467 for (ssize_t i = 0; i < numToLaunch + useCallingThread; ++i) {
468 states.emplace_back(defaultState());
469 }
470
471 if (numToLaunch == 1 && !useCallingThread) {
472 taskSet.schedule(
473 [&s = states.front(), range, f = std::move(f)]() { f(s, range.start, range.end); });
474 if (options.wait) {
475 taskSet.wait();
476 }
477 return;
478 } else if (numToLaunch == 0) {
479 f(*states.begin(), range.start, range.end);
480 if (options.wait) {
481 taskSet.wait();
482 }
483 return;
484 }
485
486 const IntegerT chunk = range.calcChunkSize(numToLaunch, useCallingThread);
487
488 if (options.wait) {
489 alignas(kCacheLineSize) std::atomic<IntegerT> index(range.start);
490 auto worker = [end = range.end, &index, f, chunk](auto& s) {
491 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
492
493 while (true) {
494 IntegerT cur = index.fetch_add(chunk, std::memory_order_relaxed);
495 if (cur >= end) {
496 break;
497 }
498 f(s, cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
499 }
500 };
501
502 auto it = states.begin();
503 for (ssize_t i = 0; i < numToLaunch; ++i) {
504 taskSet.schedule([&s = *it++, worker]() { worker(s); });
505 }
506 worker(*it);
507 taskSet.wait();
508 } else {
509 struct Atomic {
510 Atomic(IntegerT i) : index(i) {}
511 char buffer[kCacheLineSize];
512 std::atomic<IntegerT> index;
513 char buffer2[kCacheLineSize];
514 };
515 auto wrapper = std::make_shared<Atomic>(range.start);
516 auto worker = [end = range.end, wrapper = std::move(wrapper), f, chunk](auto& s) {
517 auto recurseInfo = detail::PerPoolPerThreadInfo::parForRecurse();
518 while (true) {
519 IntegerT cur = wrapper->index.fetch_add(chunk, std::memory_order_relaxed);
520 if (cur >= end) {
521 break;
522 }
523 f(s, cur, std::min<IntegerT>(static_cast<IntegerT>(cur + chunk), end));
524 }
525 };
526
527 auto it = states.begin();
528 for (ssize_t i = 0; i < numToLaunch; ++i) {
529 taskSet.schedule([&s = *it++, worker]() { worker(s); }, ForceQueuingTag());
