180template <typename F>
181inline void ThreadPool::schedule(F&& f) {
182 ssize_t curWork = workRemaining_.load(std::memory_order_relaxed);
183 ssize_t quickLoadFactor = numThreads_.load(std::memory_order_relaxed);
184 quickLoadFactor += quickLoadFactor / 2;
185 if ((detail::PerPoolPerThreadInfo::isPoolRecursive(this) && curWork > quickLoadFactor) ||
186 (curWork > poolLoadFactor_.load(std::memory_order_relaxed))) {
187 f();
188 } else {
189 schedule(std::move(f), ForceQueuingTag());
190 }
191}
