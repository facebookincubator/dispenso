195 void conditionallyWake() {
196 if (enableEpochWaiter_.load(std::memory_order_acquire)) {
197 // A rare race to overwake is preferable to a race that underwakes.
198 auto queuedWork = queuedWork_.fetch_add(1, std::memory_order_acq_rel) + 1;
199 auto idle = idleButAwake_.load(std::memory_order_acquire);
200 if (idle < queuedWork) {
201 wake();
202 }
203 }
204 }

259template <typename F>
260inline void ThreadPool::schedule(F&& f) {
261 ssize_t curWork = workRemaining_.load(std::memory_order_relaxed);
262 ssize_t quickLoadFactor = numThreads_.load(std::memory_order_relaxed);
263 quickLoadFactor += quickLoadFactor / 2;
264 if ((detail::PerPoolPerThreadInfo::isPoolRecursive(this) && curWork > quickLoadFactor) ||
265 (curWork > poolLoadFactor_.load(std::memory_order_relaxed))) {
266 f();
267 } else {
268 schedule(std::forward<F>(f), ForceQueuingTag());
269 }
270}
271
272template <typename F>
273inline void ThreadPool::schedule(F&& f, ForceQueuingTag) {
274 if (auto* token =
275 static_cast<moodycamel::ProducerToken*>(detail::PerPoolPerThreadInfo::producer(this))) {
276 schedule(*token, std::forward<F>(f), ForceQueuingTag());
277 return;
278 }
279
280 if (!numThreads_.load(std::memory_order_relaxed)) {
281 f();
282 return;
283 }
284 workRemaining_.fetch_add(1, std::memory_order_release);
285 DISPENSO_TSAN_ANNOTATE_IGNORE_WRITES_BEGIN();
286 bool enqueued = work_.enqueue({std::forward<F>(f)});
287 DISPENSO_TSAN_ANNOTATE_IGNORE_WRITES_END();
288 (void)(enqueued); // unused
289 assert(enqueued);
290
291 conditionallyWake();
292}
